1388static inline int can_mcan_sys_read_mram(mem_addr_t base, uint16_t offset, void *dst, size_t len)
1389{
1390 volatile uint32_t *src32 = (volatile uint32_t *)(base + offset);
1391 uint32_t *dst32 = (uint32_t *)dst;
1392 size_t len32 = len / sizeof(uint32_t);
1393
1394 __ASSERT(base % 4U == 0U, "base must be a multiple of 4");
1395 __ASSERT(offset % 4U == 0U, "offset must be a multiple of 4");
1396 __ASSERT(POINTER_TO_UINT(dst) % 4U == 0U, "dst must be 32-bit aligned");
1397 __ASSERT(len % 4U == 0U, "len must be a multiple of 4");
1398
1399#if defined(CONFIG_CACHE_MANAGEMENT) && defined(CONFIG_DCACHE)
1400 int err;
1401
1402 err = sys_cache_data_invd_range((void *)(base + offset), len);
1403 if (err != 0) {
1404 return err;
1405 }
1406#endif /* !defined(CONFIG_CACHE_MANAGEMENT) && defined(CONFIG_DCACHE) */
1407
1408 while (len32-- > 0) {
1409 *dst32++ = *src32++;
1410 }
1411
1412 return 0;
1413}
