43/* Maximum and minimum value TIME_T can hold */
44#define SYS_TIME_T_MAX ((((time_t)1 << (8 * sizeof(time_t) - 2)) - 1) * 2 + 1)
45#define SYS_TIME_T_MIN (-SYS_TIME_T_MAX - 1)
46
47/* Converts ticks to seconds, discarding any fractional seconds */
48#define SYS_TICKS_TO_SECS(ticks) \
49 (((uint64_t)(ticks) >= (uint64_t)K_TICKS_FOREVER) ? SYS_TIME_T_MAX \
50 : k_ticks_to_sec_floor64(ticks))
51
52/* Converts ticks to nanoseconds, modulo NSEC_PER_SEC */
53#define SYS_TICKS_TO_NSECS(ticks) \
54 (((uint64_t)(ticks) >= (uint64_t)K_TICKS_FOREVER) \
55 ? (NSEC_PER_SEC - 1) \
56 : k_ticks_to_ns_floor64((uint64_t)(ticks) % CONFIG_SYS_CLOCK_TICKS_PER_SEC))
57
58/* Define a timespec */
59#define SYS_TIMESPEC(sec, nsec) \
60 ((struct timespec){ \
61 .tv_sec = (time_t)CLAMP((int64_t)(sec), SYS_TIME_T_MIN, SYS_TIME_T_MAX), \
62 .tv_nsec = (long)(nsec), \
63 })
64
65/* Initialize a struct timespec object from a tick count */
66#define SYS_TICKS_TO_TIMESPEC(ticks) SYS_TIMESPEC(SYS_TICKS_TO_SECS(ticks), \
67 SYS_TICKS_TO_NSECS(ticks))

69/* The semantic equivalent of K_NO_WAIT but expressed as a timespec object*/
70#define SYS_TIMESPEC_NO_WAIT SYS_TICKS_TO_TIMESPEC(0)

379static inline bool timespec_is_valid(const struct timespec *ts)
380{
381 __ASSERT_NO_MSG(ts != NULL);
382
383 return (ts->tv_nsec >= 0) && (ts->tv_nsec < (long)NSEC_PER_SEC);
384}

439static inline bool timespec_add(struct timespec *a, const struct timespec *b)
440{
441 __ASSERT_NO_MSG((a != NULL) && timespec_is_valid(a));
442 __ASSERT_NO_MSG((b != NULL) && timespec_is_valid(b));
443
444#if defined(CONFIG_SPEED_OPTIMIZATIONS) && HAS_BUILTIN(__builtin_add_overflow)
445
446 return !__builtin_add_overflow(a->tv_sec, b->tv_sec, &a->tv_sec) &&
447 !__builtin_add_overflow(a->tv_nsec, b->tv_nsec, &a->tv_nsec) &&
448 timespec_normalize(a);
449
450#else
451
452 if ((a->tv_sec < 0) && (b->tv_sec < 0) && (SYS_TIME_T_MIN - a->tv_sec > b->tv_sec)) {
453 /* negative integer overflow would occur */
454 return false;
455 }
456
457 if ((a->tv_sec > 0) && (b->tv_sec > 0) && (SYS_TIME_T_MAX - a->tv_sec < b->tv_sec)) {
458 /* positive integer overflow would occur */
459 return false;
460 }
461
462 a->tv_sec += b->tv_sec;
463 a->tv_nsec += b->tv_nsec;
464
465 return timespec_normalize(a);
466
467#endif
468}

482static inline bool timespec_negate(struct timespec *ts)
483{
484 __ASSERT_NO_MSG((ts != NULL) && timespec_is_valid(ts));
485
486#if defined(CONFIG_SPEED_OPTIMIZATIONS) && HAS_BUILTIN(__builtin_sub_overflow)
487
488 return !__builtin_sub_overflow(0LL, ts->tv_sec, &ts->tv_sec) &&
489 !__builtin_sub_overflow(0L, ts->tv_nsec, &ts->tv_nsec) && timespec_normalize(ts);
490
491#else
492
493 if (ts->tv_sec == SYS_TIME_T_MIN) {
494 /* -SYS_TIME_T_MIN > SYS_TIME_T_MAX, so positive integer overflow would occur */
495 return false;
496 }
497
498 ts->tv_sec = -ts->tv_sec;
499 ts->tv_nsec = -ts->tv_nsec;
500
501 return timespec_normalize(ts);
502
503#endif
504}

521static inline bool timespec_sub(struct timespec *a, const struct timespec *b)
522{
523 __ASSERT_NO_MSG(a != NULL);
524 __ASSERT_NO_MSG(b != NULL);
525
526 struct timespec neg = *b;
527
528 return timespec_negate(&neg) && timespec_add(a, &neg);

543static inline int timespec_compare(const struct timespec *a, const struct timespec *b)
544{
545 __ASSERT_NO_MSG((a != NULL) && timespec_is_valid(a));
546 __ASSERT_NO_MSG((b != NULL) && timespec_is_valid(b));
547
548 return (((a->tv_sec == b->tv_sec) && (a->tv_nsec < b->tv_nsec)) * -1) +
549 (((a->tv_sec == b->tv_sec) && (a->tv_nsec > b->tv_nsec)) * 1) +
550 ((a->tv_sec < b->tv_sec) * -1) + ((a->tv_sec > b->tv_sec));
551}

594static inline void timespec_from_timeout(k_timeout_t timeout, struct timespec *ts)
595{
596 __ASSERT_NO_MSG(ts != NULL);
597 __ASSERT_NO_MSG(Z_IS_TIMEOUT_RELATIVE(timeout) ||
598 (IS_ENABLED(CONFIG_TIMEOUT_64BIT) &&
599 K_TIMEOUT_EQ(timeout, (k_timeout_t){K_TICKS_FOREVER})));
600
601 /* equivalent of K_FOREVER without including kernel.h */
602 if (K_TIMEOUT_EQ(timeout, (k_timeout_t){K_TICKS_FOREVER})) {
603 /* duration == K_TICKS_FOREVER ticks */
604 *ts = SYS_TIMESPEC_FOREVER;
605 /* equivalent of K_NO_WAIT without including kernel.h */
606 } else if (K_TIMEOUT_EQ(timeout, (k_timeout_t){0})) {
607 /* duration <= 0 ticks */
608 *ts = SYS_TIMESPEC_NO_WAIT;
609 } else {
610 *ts = SYS_TICKS_TO_TIMESPEC(timeout.ticks);
611 }
612
613 __ASSERT_NO_MSG(timespec_is_valid(ts));
614}

645static inline k_timeout_t timespec_to_timeout(const struct timespec *req, struct timespec *rem)
646{
647 k_timeout_t timeout;
648 struct timespec temp = SYS_TIMESPEC_NO_WAIT;
649
650 __ASSERT_NO_MSG((req != NULL) && timespec_is_valid(req));
651
652 if (timespec_compare(req, &temp) <= 0) {
653 if (rem != NULL) {
654 *rem = *req;
655 }
656 /* equivalent of K_NO_WAIT without including kernel.h */
657 timeout.ticks = 0;
658 return timeout;
659 }
660
661 temp = SYS_TIMESPEC_FOREVER;
662
663 if (timespec_compare(req, &temp) == 0) {
664 if (rem != NULL) {
665 *rem = SYS_TIMESPEC_NO_WAIT;
666 }
667 /* equivalent of K_FOREVER without including kernel.h */
668 timeout.ticks = K_TICKS_FOREVER;
669 return timeout;
670 }
671
672 temp = SYS_TIMESPEC_MAX;
673
674 if (timespec_compare(req, &temp) >= 0) {
675 /* round down to align to max ticks */
676 timeout.ticks = K_TICK_MAX;
677 } else {
678 /* round up to align to next tick boundary */
679 timeout.ticks = CLAMP(k_ns_to_ticks_ceil64(req->tv_nsec) +
680 k_sec_to_ticks_ceil64(req->tv_sec),
681 K_TICK_MIN, K_TICK_MAX);
682 }
683
684 if (rem != NULL) {
685 timespec_from_timeout(timeout, rem);
686 timespec_sub(rem, req);
687 timespec_negate(rem);
688 }
689
690 return timeout;
691}
