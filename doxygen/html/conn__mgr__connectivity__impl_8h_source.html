337static inline bool conn_mgr_binding_get_flag(struct conn_mgr_conn_binding *binding,
338 enum conn_mgr_if_flag flag)
339{
340 bool value = false;
341
342 conn_mgr_binding_lock(binding);

344 value = !!(binding->flags & BIT(flag));
