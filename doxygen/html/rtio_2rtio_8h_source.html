156static inline size_t rtio_mempool_block_size(const struct rtio *r)
157{
158#ifndef CONFIG_RTIO_SYS_MEM_BLOCKS
159 ARG_UNUSED(r);
160 return 0;
161#else
162 if (r == NULL || r->block_pool == NULL) {
163 return 0;
164 }
165 return BIT(r->block_pool->info.blk_sz_shift);
166#endif
167}

176#if defined(CONFIG_RTIO_SYS_MEM_BLOCKS) || defined(__DOXYGEN__)
177static inline uint16_t __rtio_compute_mempool_block_index(const struct rtio *r, const void *ptr)
178{
179 uintptr_t addr = (uintptr_t)ptr;
180 struct sys_mem_blocks *mem_pool = r->block_pool;
181 uint32_t block_size = rtio_mempool_block_size(r);
182
183 uintptr_t buff = (uintptr_t)mem_pool->buffer;
184 uint32_t buff_size = mem_pool->info.num_blocks * block_size;
185
186 if (addr < buff || addr >= buff + buff_size) {
187 return UINT16_MAX;
188 }
189 return (addr - buff) / block_size;
190}
191#endif

193static inline int rtio_block_pool_alloc(struct rtio *r, size_t min_sz,
194 size_t max_sz, uint8_t **buf, uint32_t *buf_len)
195{
196#ifndef CONFIG_RTIO_SYS_MEM_BLOCKS
197 ARG_UNUSED(r);
198 ARG_UNUSED(min_sz);
199 ARG_UNUSED(max_sz);
200 ARG_UNUSED(buf);
201 ARG_UNUSED(buf_len);
202 return -ENOTSUP;
203#else
204 const uint32_t block_size = rtio_mempool_block_size(r);
205 uint32_t bytes = max_sz;
206
207 /* Not every context has a block pool and the block size may return 0 in
208 * that case
209 */
210 if (block_size == 0) {
211 return -ENOMEM;
212 }
213
214 do {
215 size_t num_blks = DIV_ROUND_UP(bytes, block_size);
216 int rc = sys_mem_blocks_alloc_contiguous(r->block_pool, num_blks, (void **)buf);
217
218 if (rc == 0) {
219 *buf_len = num_blks * block_size;
220 return 0;
221 }
222
223 if (bytes <= block_size) {
224 break;
225 }
226
227 bytes -= block_size;
228 } while (bytes >= min_sz);
229
230 return -ENOMEM;
231#endif
232}

234static inline void rtio_block_pool_free(struct rtio *r, void *buf, uint32_t buf_len)
235{
236#ifndef CONFIG_RTIO_SYS_MEM_BLOCKS
237 ARG_UNUSED(r);
238 ARG_UNUSED(buf);
239 ARG_UNUSED(buf_len);
240#else
241 size_t num_blks = buf_len >> r->block_pool->info.blk_sz_shift;
242
243 sys_mem_blocks_free_contiguous(r->block_pool, buf, num_blks);
244#endif
245}

327static inline struct rtio_sqe *rtio_sqe_acquire(struct rtio *r)
328{
329 SYS_PORT_TRACING_FUNC_ENTER(rtio, sqe_acquire, r);
330 struct rtio_iodev_sqe *iodev_sqe = rtio_sqe_pool_alloc(r->sqe_pool);
331
332 if (iodev_sqe == NULL) {
333 SYS_PORT_TRACING_FUNC_EXIT(rtio, sqe_acquire, r, NULL);
334 return NULL;
335 }
336
337 mpsc_push(&r->sq, &iodev_sqe->q);
338
339 SYS_PORT_TRACING_FUNC_EXIT(rtio, sqe_acquire, r, &iodev_sqe->sqe);
340 return &iodev_sqe->sqe;
341}

355static inline int rtio_sqe_acquire_array(struct rtio *r, size_t n, struct rtio_sqe **sqes)
356{
357 struct rtio_iodev_sqe *iodev_sqe;
358 size_t i;
359
360 for (i = 0; i < n; i++) {
361 iodev_sqe = rtio_sqe_pool_alloc(r->sqe_pool);
362 if (iodev_sqe == NULL) {
363 break;
364 }
365 sqes[i] = &iodev_sqe->sqe;
366 }
367
368 /* Not enough SQEs in the pool */
369 if (i < n) {
370 while (i > 0) {
371 i--;
372 iodev_sqe = CONTAINER_OF(sqes[i], struct rtio_iodev_sqe, sqe);
373 rtio_sqe_pool_free(r->sqe_pool, iodev_sqe);
374 sqes[i] = NULL;
375 }
376
377 return -ENOMEM;
378 }
379
380 for (i = 0; i < n; i++) {
381 iodev_sqe = CONTAINER_OF(sqes[i], struct rtio_iodev_sqe, sqe);
382 mpsc_push(&r->sq, &iodev_sqe->q);
383 }
384
385 return 0;
386}

394{
395 struct rtio_iodev_sqe *iodev_sqe;
396 struct mpsc_node *node = mpsc_pop(&r->sq);

443static inline struct rtio_cqe *rtio_cqe_consume(struct rtio *r)
444{
445 SYS_PORT_TRACING_FUNC_ENTER(rtio, cqe_consume, r);
446 struct mpsc_node *node;
447 struct rtio_cqe *cqe = NULL;
448
449#ifdef CONFIG_RTIO_CONSUME_SEM
450 if (k_sem_take(r->consume_sem, K_NO_WAIT) != 0) {
451 SYS_PORT_TRACING_FUNC_EXIT(rtio, cqe_consume, r, NULL);
452 return NULL;
453 }
454#endif
455
456 node = mpsc_pop(&r->cq);
457 if (node == NULL) {
458 SYS_PORT_TRACING_FUNC_EXIT(rtio, cqe_consume, r, NULL);
459 return NULL;
460 }
461 cqe = CONTAINER_OF(node, struct rtio_cqe, q);
462
463 SYS_PORT_TRACING_FUNC_EXIT(rtio, cqe_consume, r, cqe);
464 return cqe;
465}

477static inline struct rtio_cqe *rtio_cqe_consume_block(struct rtio *r)
478{
479 struct mpsc_node *node;
480 struct rtio_cqe *cqe;
481
482#ifdef CONFIG_RTIO_CONSUME_SEM
483 k_sem_take(r->consume_sem, K_FOREVER);

515static inline int rtio_flush_completion_queue(struct rtio *r)
516{
517 struct rtio_cqe *cqe;
518 int res = 0;
519
520 do {
521 cqe = rtio_cqe_consume(r);
522 if (cqe != NULL) {
523 if ((cqe->result < 0) && (res == 0)) {
524 res = cqe->result;
525 }
526 rtio_cqe_release(r, cqe);
527 }
528 } while (cqe != NULL);
529
530 return res;
531}

561static inline uint32_t rtio_cqe_compute_flags(struct rtio_iodev_sqe *iodev_sqe)
562{
563 uint32_t flags = 0;
564
565#ifdef CONFIG_RTIO_SYS_MEM_BLOCKS
566 if (iodev_sqe->sqe.op == RTIO_OP_RX && iodev_sqe->sqe.flags & RTIO_SQE_MEMPOOL_BUFFER) {
567 struct rtio *r = iodev_sqe->r;
568 struct sys_mem_blocks *mem_pool = r->block_pool;
569 unsigned int blk_index = 0;
570 unsigned int blk_count = 0;
571
572 if (iodev_sqe->sqe.rx.buf) {
573 blk_index = (iodev_sqe->sqe.rx.buf - mem_pool->buffer) >>
574 mem_pool->info.blk_sz_shift;
575 blk_count = iodev_sqe->sqe.rx.buf_len >> mem_pool->info.blk_sz_shift;
576 }
577 flags = RTIO_CQE_FLAG_PREP_MEMPOOL(blk_index, blk_count);
578 }
579#else
580 ARG_UNUSED(iodev_sqe);
581#endif
582
583 return flags;
584}

604static inline int z_impl_rtio_cqe_get_mempool_buffer(const struct rtio *r, struct rtio_cqe *cqe,
605 uint8_t **buff, uint32_t *buff_len)
606{
607#ifdef CONFIG_RTIO_SYS_MEM_BLOCKS
608 if (RTIO_CQE_FLAG_GET(cqe->flags) == RTIO_CQE_FLAG_MEMPOOL_BUFFER) {
609 unsigned int blk_idx = RTIO_CQE_FLAG_MEMPOOL_GET_BLK_IDX(cqe->flags);
610 unsigned int blk_count = RTIO_CQE_FLAG_MEMPOOL_GET_BLK_CNT(cqe->flags);
611 uint32_t blk_size = rtio_mempool_block_size(r);
612
613 *buff_len = blk_count * blk_size;
614
615 if (blk_count > 0) {
616 *buff = r->block_pool->buffer + blk_idx * blk_size;
617
618 __ASSERT_NO_MSG(*buff >= r->block_pool->buffer);
619 __ASSERT_NO_MSG(*buff <
620 r->block_pool->buffer + blk_size * r->block_pool->info.num_blocks);
621 } else {
622 *buff = NULL;
623 }
624 return 0;
625 }
626 return -EINVAL;
627#else
628 ARG_UNUSED(r);
629 ARG_UNUSED(cqe);
630 ARG_UNUSED(buff);
631 ARG_UNUSED(buff_len);
632
633 return -ENOTSUP;
634#endif
635}

678static inline void rtio_cqe_submit(struct rtio *r, int result, void *userdata, uint32_t flags)
679{
680 SYS_PORT_TRACING_FUNC_ENTER(rtio, cqe_submit, r, result, flags);
681 struct rtio_cqe *cqe = rtio_cqe_acquire(r);
682
683 if (cqe == NULL) {
684 atomic_inc(&r->xcqcnt);
685 } else {
686 cqe->result = result;
687 cqe->userdata = userdata;
688 cqe->flags = flags;
689 rtio_cqe_produce(r, cqe);
690#ifdef CONFIG_RTIO_CONSUME_SEM
691 k_sem_give(r->consume_sem);
692#endif
693 }
694
695 /* atomic_t isn't guaranteed to wrap correctly as it could be signed, so
696 * we must resort to a cas loop.
697 */
698 atomic_t val, new_val;
699
700 do {
701 val = atomic_get(&r->cq_count);
702 new_val = (atomic_t)((uintptr_t)val + 1);
703 } while (!atomic_cas(&r->cq_count, val, new_val));
704
705#ifdef CONFIG_RTIO_SUBMIT_SEM
706 if (r->submit_count > 0) {
707 r->submit_count--;
708 if (r->submit_count == 0) {
709 k_sem_give(r->submit_sem);
710 }
711 }
712#endif
713 SYS_PORT_TRACING_FUNC_EXIT(rtio, cqe_submit, r);
714}

730static inline int rtio_sqe_rx_buf(const struct rtio_iodev_sqe *iodev_sqe, uint32_t min_buf_len,
731 uint32_t max_buf_len, uint8_t **buf, uint32_t *buf_len)
732{
733 struct rtio_sqe *sqe = (struct rtio_sqe *)&iodev_sqe->sqe;
734
735#ifdef CONFIG_RTIO_SYS_MEM_BLOCKS
736 if (sqe->op == RTIO_OP_RX && sqe->flags & RTIO_SQE_MEMPOOL_BUFFER) {
737 struct rtio *r = iodev_sqe->r;
738
739 if (sqe->rx.buf != NULL) {
740 if (sqe->rx.buf_len < min_buf_len) {
741 return -ENOMEM;
742 }
743 *buf = sqe->rx.buf;
744 *buf_len = sqe->rx.buf_len;
745 return 0;
746 }
747
748 int rc = rtio_block_pool_alloc(r, min_buf_len, max_buf_len, buf, buf_len);
749 if (rc == 0) {
750 sqe->rx.buf = *buf;
751 sqe->rx.buf_len = *buf_len;
752 return 0;
753 }
754
755 return -ENOMEM;
756 }
757#else
758 ARG_UNUSED(max_buf_len);
759#endif
760
761 if (sqe->rx.buf_len < min_buf_len) {
762 return -ENOMEM;
763 }
764
765 *buf = sqe->rx.buf;
766 *buf_len = sqe->rx.buf_len;
767 return 0;
768}

880__syscall int rtio_sqe_copy_in_get_handles(struct rtio *r, const struct rtio_sqe *sqes,
881 struct rtio_sqe **handle, size_t sqe_count);
882
883static inline int z_impl_rtio_sqe_copy_in_get_handles(struct rtio *r, const struct rtio_sqe *sqes,
884 struct rtio_sqe **handle,
885 size_t sqe_count)
886{
887 struct rtio_sqe *sqe;
888 uint32_t acquirable = rtio_sqe_acquirable(r);
889
890 if (acquirable < sqe_count) {
891 return -ENOMEM;
892 }
893
894 for (unsigned long i = 0; i < sqe_count; i++) {
895 sqe = rtio_sqe_acquire(r);
896 __ASSERT_NO_MSG(sqe != NULL);
897 if (handle != NULL && i == 0) {
898 *handle = sqe;
899 }
900 *sqe = sqes[i];
901 }
902
903 return 0;
