1184static inline int bt_obex_add_header_body_or_end_body(struct net_buf *buf, uint16_t mopl,
1185 uint16_t len, const uint8_t *body,
1186 uint16_t *added_len)
1187{
1188 uint16_t tx_len;
1189 int err;
1190
1191 /*
1192 * OBEX Version 1.5, section 2.2.9 Body, End-of-Body
1193 * The `body` could be a NULL, so the `len` of the name could 0.
1194 * In some cases, the object body data is generated on the fly and the end cannot
1195 * be anticipated, so it is legal to send a zero length End-of-Body header.
1196 */
1197 if ((buf == NULL) || ((len != 0) && (body == NULL)) || (added_len == NULL) ||
1198 (mopl < BT_OBEX_MIN_MTU)) {
1199 return -EINVAL;
1200 }
1201
1202 tx_len = BT_OBEX_PDU_LEN(mopl);
1203 if (tx_len <= buf->len) {
1204 return -ENOMEM;
1205 }
1206
1207 *added_len = 0;
1208
1209 tx_len = MIN((tx_len - buf->len), net_buf_tailroom(buf));
1210 if (tx_len <= BT_OBEX_HDR_LEN_OF_HEADER_BODY) {
1211 return 0;
1212 }
1213
1214 tx_len = BT_OBEX_DATA_LEN_OF_HEADER_BODY(tx_len);
1215 if (tx_len >= len) {
1216 *added_len = len;
1217 err = bt_obex_add_header_end_body(buf, len, body);
1218 } else {
1219 *added_len = tx_len;
1220 err = bt_obex_add_header_body(buf, tx_len, body);
1221 }
1222
1223 return err;
1224}
