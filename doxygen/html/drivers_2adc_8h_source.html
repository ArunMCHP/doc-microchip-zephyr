1230static inline int adc_stream(struct rtio_iodev *iodev, struct rtio *ctx, void *userdata,
1231 struct rtio_sqe **handle)
1232{
1233 if (IS_ENABLED(CONFIG_USERSPACE)) {
1234 struct rtio_sqe sqe;
1235
1236 rtio_sqe_prep_read_multishot(&sqe, iodev, RTIO_PRIO_NORM, userdata);
1237 rtio_sqe_copy_in_get_handles(ctx, &sqe, handle, 1);
1238 } else {
1239 struct rtio_sqe *sqe = rtio_sqe_acquire(ctx);
1240
1241 if (sqe == NULL) {
1242 return -ENOMEM;
1243 }
1244 if (handle != NULL) {
1245 *handle = sqe;
1246 }
1247 rtio_sqe_prep_read_multishot(sqe, iodev, RTIO_PRIO_NORM, userdata);
1248 }
1249 rtio_submit(ctx, 0);
1250 return 0;
1251}

1315static inline int adc_raw_to_millivolts(int32_t ref_mv, enum adc_gain gain, uint8_t resolution,
1316 int32_t *valp)
1317{
1318 int64_t adc_mv = (int64_t)*valp * (int64_t)ref_mv;
1319 int ret = adc_gain_invert_64(gain, &adc_mv);
1320
1321 if (ret == 0) {
1322 adc_mv = adc_mv >> resolution;
1323 if (adc_mv > INT32_MAX || adc_mv < INT32_MIN) {
1324 __ASSERT_MSG_INFO("conversion result is out of range");
1325 }
1326
1327 *valp = (int32_t)adc_mv;
1328 }
1329
1330 return ret;
1331}

1338static inline int adc_raw_to_microvolts(int32_t ref_mv, enum adc_gain gain, uint8_t resolution,
1339 int32_t *valp)
1340{
1341 int64_t adc_uv = (int64_t)*valp * ref_mv * 1000;
1342 int ret = adc_gain_invert_64(gain, &adc_uv);
1343
1344 if (ret == 0) {
1345 *valp = (int32_t)(adc_uv >> resolution);
1346 }
1347
1348 return ret;
1349}

1367static inline int adc_raw_to_x_dt_chan(adc_raw_to_x_fn conv_func,
1368 const struct adc_dt_spec *spec,
1369 const struct adc_channel_cfg *channel_cfg,
1370 int32_t *valp)
1371{
1372 int32_t vref_mv;
1373 uint8_t resolution;
1374
1375 if (!spec->channel_cfg_dt_node_exists) {
1376 return -ENOTSUP;
1377 }
1378
1379 if (channel_cfg->reference == ADC_REF_INTERNAL) {
1380 vref_mv = (int32_t)adc_ref_internal(spec->dev);
1381 } else {
1382 vref_mv = spec->vref_mv;
1383 }
1384
1385 resolution = spec->resolution;
1386
1387 /*
1388 * For differential channels, one bit less needs to be specified
1389 * for resolution to achieve correct conversion.
1390 */
1391 if (channel_cfg->differential) {
1392 resolution -= 1U;
1393 }
1394
1395 return conv_func(vref_mv, channel_cfg->gain, resolution, valp);
1396}
