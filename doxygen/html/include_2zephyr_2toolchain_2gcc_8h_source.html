135/* Unaligned access */
136#define UNALIGNED_GET(g) \
137__extension__ ({ \
138 struct __attribute__((__packed__)) { \
139 __typeof__(*(g)) __v; \
140 } *__g = (__typeof__(__g)) (g); \
141 __g->__v; \
142})

147/* Version of UNALIGNED_PUT() which issues a compiler_barrier() after
148 * the store. It is required to workaround an apparent optimization
149 * bug in GCC for ARM Cortex-M3 and higher targets, when multiple
150 * byte, half-word and word stores (strb, strh, str instructions),
151 * which support unaligned access, can be coalesced into store double
152 * (strd) instruction, which doesn't support unaligned access (the
153 * compilers in question do this optimization ignoring __packed__
154 * attribute).
155 */
156#define UNALIGNED_PUT(v, p) \
157do { \
158 struct __attribute__((__packed__)) { \
159 __typeof__(*p) __v; \
160 } *__p = (__typeof__(__p)) (p); \
161 __p->__v = (v); \
162 compiler_barrier(); \
163} while (false)
164
165#else
166
167#define UNALIGNED_PUT(v, p) \
168do { \
169 struct __attribute__((__packed__)) { \
170 __typeof__(*p) __v; \
171 } *__p = (__typeof__(__p)) (p); \
172 __p->__v = (v); \
173} while (false)
174
175#endif

177/*
178 * Get the address of a structure member even if the member may not be properly
179 * aligned. Note that accessing such an address must be done with care (for
180 * example with UNALIGNED_GET/PUT) and cannot be in general de-referenced to
181 * access the member directly, as that would cause a fault in architectures
182 * which have alignment requirements.
183 */
184#define UNALIGNED_MEMBER_ADDR(_p, _member) ((__typeof__(_p->_member) *) \
185 (((intptr_t)(_p)) + offsetof(__typeof__(*_p), _member)))

443/*
444 * These macros specify the section in which a given function or variable
445 * resides.
446 *
447 * - SECTION_FUNC allows only one function to reside in a sub-section
448 * - SECTION_SUBSEC_FUNC allows multiple functions to reside in a sub-section
449 * This ensures that garbage collection only discards the section
450 * if all functions in the sub-section are not referenced.
451 */
452
453#if defined(CONFIG_ARC)
454/*
455 * Need to use assembly macros because ';' is interpreted as the start of
456 * a single line comment in the ARC assembler.
457 *
458 * Also, '\‍()' is needed in the .section directive of these macros for
459 * correct substitution of the 'section' variable.
460 */
461
462.macro section_var section, symbol
463 .section .\section\‍().\symbol
464 \symbol :
465.endm
466
467.macro section_func section, symbol
468 .section .\section\().\symbol, "ax"
469 FUNC_CODE()
470 PERFOPT_ALIGN
471 \symbol :
472 FUNC_INSTR(\symbol)
473.endm
474
475.macro section_subsec_func section, subsection, symbol
476 .section .\section\().\subsection, "ax"
477 PERFOPT_ALIGN
478 \symbol :
479.endm
480
481#define SECTION_VAR(sect, sym) section_var sect, sym
482#define SECTION_FUNC(sect, sym) section_func sect, sym
483#define SECTION_SUBSEC_FUNC(sect, subsec, sym) \
484 section_subsec_func sect, subsec, sym
485#else /* !CONFIG_ARC */
486
487#define SECTION_VAR(sect, sym) .section .sect.sym; sym:
488#define SECTION_FUNC(sect, sym) \
489 .section .sect.sym, "ax"; \
490 FUNC_CODE() \
491 PERFOPT_ALIGN; sym : \
492 FUNC_INSTR(sym)
493#define SECTION_SUBSEC_FUNC(sect, subsec, sym) \
494 .section .sect.subsec, "ax"; PERFOPT_ALIGN; sym :
495
496#endif /* CONFIG_ARC */
497
498#endif /* _ASMLANGUAGE */

528/*
529 * Note that GEN_ABSOLUTE_SYM(), depending on the architecture
530 * and toolchain, may restrict the range of values permitted
531 * for assignment to the named symbol.
532 *
533 * For example, on x86, "value" is interpreted as signed
534 * 32-bit integer. Passing in an unsigned 32-bit integer
535 * with MSB set would result in a negative integer.
536 * Moreover, GCC would error out if an integer larger
537 * than 2^32-1 is passed as "value".
538 */
539
540/*
541 * GEN_ABSOLUTE_SYM_KCONFIG() is outputted by the build system
542 * to generate named symbol/value pairs for kconfigs.
543 */
544
545#if defined(CONFIG_ARM) || (defined(CONFIG_ARCH_POSIX) && defined(__arm__))
546
547/*
548 * GNU/ARM backend does not have a proper operand modifier which does not
549 * produces prefix # followed by value, such as %0 for PowerPC, Intel, and
550 * MIPS. The workaround performed here is using %B0 which converts
551 * the value to ~(value). Thus "n"(~(value)) is set in operand constraint
552 * to output (value) in the ARM specific GEN_OFFSET macro.
553 */
554
555#define GEN_ABSOLUTE_SYM(name, value) \
556 __asm__ __volatile__(".globl\t" #name "\n\t.equ\t" #name \
557 ",%B0" \
558 "\n\t.type\t" #name ",%%object" : : "n"(~(value)))
559
560#define GEN_ABSOLUTE_SYM_KCONFIG(name, value) \
561 __asm__ __volatile__(".globl\t" #name \
562 "\n\t.equ\t" #name "," #value \
563 "\n\t.type\t" #name ",%object")
564
565#elif defined(CONFIG_ARC) || defined(CONFIG_ARM64) \
566 || defined(CONFIG_ARCH_POSIX) /*&& (__x86_64 or __i*86 or __aarch64__)*/ \
567 || defined(CONFIG_X86)
568
569#define GEN_ABSOLUTE_SYM(name, value) \
570 __asm__ __volatile__(".globl\t" #name "\n\t.equ\t" #name \
571 ",%c0" \
572 "\n\t.type\t" #name ",@object" : : "n"(value))
573
574#define GEN_ABSOLUTE_SYM_KCONFIG(name, value) \
575 __asm__ __volatile__(".globl\t" #name \
576 "\n\t.equ\t" #name "," #value \
577 "\n\t.type\t" #name ",@object")
578
579#elif defined(CONFIG_RISCV) || defined(CONFIG_XTENSA) || \
580 defined(CONFIG_MIPS) || defined(CONFIG_OPENRISC)
581
582/* No special prefixes necessary for constants in this arch AFAICT */
