435static inline int onoff_cancel_or_release(struct onoff_manager *mgr,
436 struct onoff_client *cli)
437{
438 int rv = onoff_cancel(mgr, cli);

440 if (rv == -EALREADY) {
441 rv = onoff_release(mgr);
