458static inline int tcpc_is_cc_rp(enum tc_cc_voltage_state cc)
459{
460 return (cc == TC_CC_VOLT_RP_DEF) || (cc == TC_CC_VOLT_RP_1A5) || (cc == TC_CC_VOLT_RP_3A0);
