273/* If strings are removed from the binary then there is a risk of creating invalid
274 * cbprintf package if %p is used with character pointer which is interpreted as
275 * string. A compile time check is performed (since format string is known at
276 * compile time) and check fails logging message is not created but error is
277 * emitted instead. String check may increase compilation time so it is not
278 * always performed (could significantly increase CI time).
279 */
280#ifdef CONFIG_LOG_FMT_STRING_VALIDATE
281#define LOG_STRING_WARNING(_mode, _src, ...) \
282 Z_LOG_MSG_CREATE(UTIL_NOT(IS_ENABLED(CONFIG_USERSPACE)), _mode, \
283 Z_LOG_LOCAL_DOMAIN_ID, _src, LOG_LEVEL_ERR, NULL, 0, \
284 "char pointer used for %%p, cast to void *:\"%s\"", \
285 GET_ARG_N(1, __VA_ARGS__))
286
287#define LOG_POINTERS_VALIDATE(string_ok, ...) \
288 TOOLCHAIN_DISABLE_GCC_WARNING(TOOLCHAIN_WARNING_POINTER_ARITH); \
289 string_ok = Z_CBPRINTF_POINTERS_VALIDATE(__VA_ARGS__); \
290 TOOLCHAIN_ENABLE_GCC_WARNING(TOOLCHAIN_WARNING_POINTER_ARITH);
291#else
292#define LOG_POINTERS_VALIDATE(string_ok, ...) string_ok = true
293#define LOG_STRING_WARNING(_mode, _src, ...)
294#endif

314#define Z_LOG2(_level, _inst, _source, ...) \
315 TOOLCHAIN_DISABLE_CLANG_WARNING(TOOLCHAIN_WARNING_USED_BUT_MARKED_UNUSED) \
316 do { \
317 Z_LOG_LEVEL_ALL_CHECK_BREAK(_level, _inst, _source) \
318 if (IS_ENABLED(CONFIG_LOG_MODE_MINIMAL)) { \
319 Z_LOG_TO_PRINTK(_level, __VA_ARGS__); \
320 break; \
321 } \
322 int _mode; \
323 bool string_ok; \
324 LOG_POINTERS_VALIDATE(string_ok, __VA_ARGS__); \

327 break; \

329 Z_LOG_MSG_CREATE(UTIL_NOT(IS_ENABLED(CONFIG_USERSPACE)), _mode, \
