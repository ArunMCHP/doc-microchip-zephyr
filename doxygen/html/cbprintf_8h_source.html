520/* @internal Context used for package copying. */
521struct z_cbprintf_buf_desc {
522 void *buf;
523 size_t size;
524 size_t off;
525};
526
527/* @internal Function callback used for package copying. */
528static inline int z_cbprintf_cpy(const void *buf, size_t len, void *ctx)
529{
530 struct z_cbprintf_buf_desc *desc = (struct z_cbprintf_buf_desc *)ctx;
531
532 if (len == 0) {
533 return 0;
534 }
535
536 if ((desc->size - desc->off) < len) {
537 return -ENOSPC;
538 }
539
540 memcpy(&((uint8_t *)desc->buf)[desc->off], buf, len);
541 desc->off += len;
542
543 return len;
544}
