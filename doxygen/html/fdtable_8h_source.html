229static inline int zvfs_fdtable_call_ioctl(const struct fd_op_vtable *vtable, void *obj,
230 unsigned long request, ...)
231{
232 va_list args;
233 int res;
234
235 va_start(args, request);
236 res = vtable->ioctl(obj, request, args);
237 va_end(args);
238
239 return res;
240}
