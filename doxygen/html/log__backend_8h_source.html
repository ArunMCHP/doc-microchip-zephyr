322static inline int log_backend_format_set(const struct log_backend *backend, uint32_t log_type)
323{
324 extern size_t log_format_table_size(void);
325
326 if ((size_t)log_type >= log_format_table_size()) {
327 return -EINVAL;
328 }
329
330 if (log_format_func_t_get(log_type) == NULL) {
331 return -EINVAL;
332 }
333
334 if (backend == NULL) {
335 return -EINVAL;
336 }
337
338 if (backend->api->format_set == NULL) {
339 return -ENOTSUP;
340 }
341
342 return backend->api->format_set(backend, log_type);
343}
