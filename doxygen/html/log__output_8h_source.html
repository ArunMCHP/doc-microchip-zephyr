205static inline void log_output_write(log_output_func_t outf, uint8_t *buf, size_t len, void *ctx)
206{
207 int processed;
208
209 while (len != 0) {
210 processed = outf(buf, len, ctx);
211 len -= processed;
212 buf += processed;
213 }
214}
