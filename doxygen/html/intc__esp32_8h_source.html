13/*
14 * Interrupt allocation flags - These flags can be used to specify
15 * which interrupt qualities the code calling esp_intr_alloc* needs.
16 *
17 */
18
19/* Keep the LEVELx values as they are here; they match up with (1<<level) */
20#define ESP_INTR_FLAG_LEVEL1 (1<<1) /* Accept a Level 1 int vector, lowest priority */
21#define ESP_INTR_FLAG_LEVEL2 (1<<2) /* Accept a Level 2 int vector */
22#define ESP_INTR_FLAG_LEVEL3 (1<<3) /* Accept a Level 3 int vector */
23#define ESP_INTR_FLAG_LEVEL4 (1<<4) /* Accept a Level 4 int vector */
24#define ESP_INTR_FLAG_LEVEL5 (1<<5) /* Accept a Level 5 int vector */
25#define ESP_INTR_FLAG_LEVEL6 (1<<6) /* Accept a Level 6 int vector */
26#define ESP_INTR_FLAG_NMI (1<<7) /* Accept a Level 7 int vector, highest priority */
27#define ESP_INTR_FLAG_SHARED (1<<8) /* Interrupt can be shared between ISRs */
28#define ESP_INTR_FLAG_EDGE (1<<9) /* Edge-triggered interrupt */
29#define ESP_INTR_FLAG_IRAM (1<<10) /* ISR can be called if cache is disabled */
30#define ESP_INTR_FLAG_INTRDISABLED (1<<11) /* Return with this interrupt disabled */
31
32/* Low and medium prio interrupts. These can be handled in C. */
33#define ESP_INTR_FLAG_LOWMED (ESP_INTR_FLAG_LEVEL1|ESP_INTR_FLAG_LEVEL2|ESP_INTR_FLAG_LEVEL3)
34
35/* High level interrupts. Need to be handled in assembly. */
36#define ESP_INTR_FLAG_HIGH (ESP_INTR_FLAG_LEVEL4|ESP_INTR_FLAG_LEVEL5|ESP_INTR_FLAG_LEVEL6| \
37 ESP_INTR_FLAG_NMI)
38
39/* Mask for all level flags */
40#define ESP_INTR_FLAG_LEVELMASK (ESP_INTR_FLAG_LEVEL1|ESP_INTR_FLAG_LEVEL2|ESP_INTR_FLAG_LEVEL3| \
41 ESP_INTR_FLAG_LEVEL4|ESP_INTR_FLAG_LEVEL5|ESP_INTR_FLAG_LEVEL6| \
42 ESP_INTR_FLAG_NMI)
43
44/*
45 * Get the interrupt flags from the supplied priority.
46 */
47#define ESP_PRIO_TO_FLAGS(priority) \
48 ((priority) > 0 ? ((1 << (priority)) & ESP_INTR_FLAG_LEVELMASK) : 0)
49
50/*
51 * Check interrupt flags from input and filter unallowed values.
52 */
53#define ESP_INT_FLAGS_CHECK(int_flags) ((int_flags) & ESP_INTR_FLAG_SHARED)

55/*
56 * The esp_intr_alloc* functions can allocate an int for all *_INTR_SOURCE int sources that
57 * are routed through the interrupt mux. Apart from these sources, each core also has some internal
58 * sources that do not pass through the interrupt mux. To allocate an interrupt for these sources,
59 * pass these pseudo-sources to the functions.
60 */
61#define ETS_INTERNAL_TIMER0_INTR_SOURCE -1 /* Xtensa timer 0 interrupt source */
62#define ETS_INTERNAL_TIMER1_INTR_SOURCE -2 /* Xtensa timer 1 interrupt source */
63#define ETS_INTERNAL_TIMER2_INTR_SOURCE -3 /* Xtensa timer 2 interrupt source */
64#define ETS_INTERNAL_SW0_INTR_SOURCE -4 /* Software int source 1 */
65#define ETS_INTERNAL_SW1_INTR_SOURCE -5 /* Software int source 2 */
66#define ETS_INTERNAL_PROFILING_INTR_SOURCE -6 /* Int source for profiling */
