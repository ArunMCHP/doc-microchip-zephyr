25/*
26 * In architecture variants with non-programmable fault exceptions
27 * (e.g. Cortex-M Baseline variants), hardware ensures processor faults
28 * are given the highest interrupt priority level. SVCalls are assigned
29 * the highest configurable priority level (level 0); note, however, that
30 * this interrupt level may be shared with HW interrupts.
31 *
32 * In Cortex variants with programmable fault exception priorities we
33 * assign the highest interrupt priority level (level 0) to processor faults
34 * with configurable priority.
35 * The highest priority level may be shared with either Zero-Latency IRQs (if
36 * support for the feature is enabled) or with SVCall priority level.
37 * Regular HW IRQs are always assigned priority levels lower than the priority
38 * levels for SVCalls, Zero-Latency IRQs and processor faults.
39 *
40 * PendSV IRQ (which is used in Cortex-M variants to implement thread
41 * context-switching) is assigned the lowest IRQ priority level.
42 */
43#if defined(CONFIG_CPU_CORTEX_M_HAS_PROGRAMMABLE_FAULT_PRIOS)
44#define _EXCEPTION_RESERVED_PRIO 1
45#else
46#define _EXCEPTION_RESERVED_PRIO 0
47#endif

52 (CONFIG_ZERO_LATENCY_LEVELS), (0))

75/* Registers s16-s31 (d8-d15, q4-q7) must be preserved across subroutine calls.
