762static inline uint16_t coap_block_size_to_bytes(
763 enum coap_block_size block_size)
764{
765 return (1 << (block_size + 4));
766}
767
776static inline enum coap_block_size coap_bytes_to_block_size(uint16_t bytes)
777{
778 int sz = u32_count_trailing_zeros(bytes) - 4;
779
780 if (sz < COAP_BLOCK_16) {
781 return COAP_BLOCK_16;
782 }
783 if (sz > COAP_BLOCK_1024) {
784 return COAP_BLOCK_1024;
785 }
786 return (enum coap_block_size)sz;
787}
