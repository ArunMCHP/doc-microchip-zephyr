964static inline int hl78xx_parse_rssi(uint8_t rssi, int16_t *value)
965{
966 /* AT+CSQ returns a response +CSQ: <rssi>,<ber> where:
967 * - rssi is a integer from 0 to 31 whose values describes a signal strength
968 * between -113 dBm for 0 and -51dbM for 31 or unknown for 99
969 * - ber is an integer from 0 to 7 that describes the error rate, it can also
970 * be 99 for an unknown error rate
971 */
972 if (rssi == CSQ_RSSI_UNKNOWN) {
973 return -EINVAL;
974 }
975
976 *value = (int16_t)CSQ_RSSI_TO_DB(rssi);
977 return 0;
978}
992static inline int hl78xx_parse_rsrp(uint8_t rsrp, int16_t *value)
993{
994 /* AT+CESQ returns a response
995 * +CESQ: <rxlev>,<ber>,<rscp>,<ecn0>,<rsrq>,<rsrp> where:
996 * rsrq is a integer from 0 to 34 whose values describes the Reference
997 * Signal Receive Quality between -20 dB for 0 and -3 dB for 34
998 * (0.5 dB steps), or unknown for 255
999 * rsrp is an integer from 0 to 97 that describes the Reference Signal
1000 * Receive Power between -140 dBm for 0 and -44 dBm for 97 (1 dBm steps),
1001 * or unknown for 255
1002 */
1003 if (rsrp == CESQ_RSRP_UNKNOWN) {
1004 return -EINVAL;
1005 }
1006
1007 *value = (int16_t)CESQ_RSRP_TO_DB(rsrp);
1008 return 0;
1009}
