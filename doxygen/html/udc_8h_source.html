452static inline int udc_set_address(const struct device *dev, const uint8_t addr)
453{
454 const struct udc_api *api = (const struct udc_api *)dev->api;
455 int ret;
456
457 if (!udc_is_enabled(dev)) {
458 return -EPERM;
459 }
460
461 api->lock(dev);
462 ret = api->set_address(dev, addr);
463 api->unlock(dev);
464
465 return ret;
466}

483static inline int udc_test_mode(const struct device *dev,
484 const uint8_t mode, const bool dryrun)
485{
486 const struct udc_api *api = (const struct udc_api *)dev->api;
487 int ret;
488
489 if (!udc_is_enabled(dev)) {
490 return -EPERM;
491 }
492
493 if (api->test_mode != NULL) {
494 api->lock(dev);
495 ret = api->test_mode(dev, mode, dryrun);
496 api->unlock(dev);
497 } else {
498 ret = -ENOTSUP;
499 }
500
501 return ret;
502}
