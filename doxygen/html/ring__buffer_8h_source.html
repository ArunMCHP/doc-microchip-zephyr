174static inline void ring_buf_init(struct ring_buf *buf,
175 uint32_t size,
176 uint8_t *data)
177{
178 __ASSERT(size <= RING_BUFFER_MAX_SIZE, RING_BUFFER_SIZE_ASSERT_MSG);
179
180 buf->size = size;
181 buf->buffer = data;
182 ring_buf_internal_reset(buf, 0);
183}
