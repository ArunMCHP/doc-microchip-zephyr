276static inline int crc_verify(struct crc_ctx *ctx, crc_result_t expected)
277{
278 if (ctx == NULL) {
279 return -EINVAL;
280 }
281
282 if (ctx->state == CRC_STATE_IN_PROGRESS) {
283 return -EBUSY;
284 }
285
286 if (expected != ctx->result) {
287 return -EPERM;
288 }
289
290 return 0;
291}
