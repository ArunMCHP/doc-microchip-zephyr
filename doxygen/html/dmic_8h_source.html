238static inline uint32_t dmic_build_channel_map(uint8_t channel, uint8_t pdm,
239 enum pdm_lr lr)
240{
241 return ((((pdm & BIT_MASK(3)) << 1) | lr) <<
242 ((channel & BIT_MASK(3)) * 4U));
243}
244
257static inline void dmic_parse_channel_map(uint32_t channel_map_lo,
258 uint32_t channel_map_hi, uint8_t channel, uint8_t *pdm, enum pdm_lr *lr)
259{
260 uint32_t channel_map;
261
262 channel_map = (channel < 8) ? channel_map_lo : channel_map_hi;
263 channel_map >>= ((channel & BIT_MASK(3)) * 4U);
264
265 *pdm = (channel_map >> 1) & BIT_MASK(3);
266 *lr = (enum pdm_lr) (channel_map & BIT(0));
267}

280static inline uint32_t dmic_build_clk_skew_map(uint8_t pdm, uint8_t skew)
281{
282 return ((skew & BIT_MASK(4)) << ((pdm & BIT_MASK(3)) * 4U));
283}
