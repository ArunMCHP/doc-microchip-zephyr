12/*
13 * The NXP S32 pinmux configuration is encoded in a 32-bit field value as follows:
14 *
15 * - 0..2: Output mux Source Signal Selection (MSCR.SSS)
16 * - 3..6: Input mux Source Signal Selection (IMCR.SSS)
17 * - 7..15: Input Multiplexed Signal Configuration Register (IMCR) index
18 * - 16..24: Multiplexed Signal Configuration Register (MSCR) index
19 * - 25..27: MSCR SIUL2 instance index (0..7)
20 * - 28..30: IMCR SIUL2 instance index (0..7)
21 * - 31: Reserved for future use
22 */
23#define NXP_SIUL2_MSCR_SSS_SHIFT 0U
24#define NXP_SIUL2_MSCR_SSS_MASK BIT_MASK(3)
25#define NXP_SIUL2_IMCR_SSS_SHIFT 3U
26#define NXP_SIUL2_IMCR_SSS_MASK BIT_MASK(4)
27#define NXP_SIUL2_IMCR_IDX_SHIFT 7U
28#define NXP_SIUL2_IMCR_IDX_MASK BIT_MASK(9)
29#define NXP_SIUL2_MSCR_IDX_SHIFT 16U
30#define NXP_SIUL2_MSCR_IDX_MASK BIT_MASK(9)
31#define NXP_SIUL2_MSCR_SIUL2_IDX_SHIFT 25U
32#define NXP_SIUL2_MSCR_SIUL2_IDX_MASK BIT_MASK(3)
33#define NXP_SIUL2_IMCR_SIUL2_IDX_SHIFT 28U
34#define NXP_SIUL2_IMCR_SIUL2_IDX_MASK BIT_MASK(3)
35
