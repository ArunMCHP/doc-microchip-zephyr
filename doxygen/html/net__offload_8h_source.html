36static inline int32_t timeout_to_int32(k_timeout_t timeout)
37{
38 if (K_TIMEOUT_EQ(timeout, K_NO_WAIT)) {
39 return 0;
40 } else if (K_TIMEOUT_EQ(timeout, K_FOREVER)) {
41 return -1;
42 } else {
43 return k_ticks_to_ms_floor32(timeout.ticks);
44 }
45}
