45struct k_spinlock {
49#ifdef CONFIG_SMP
50#ifdef CONFIG_TICKET_SPINLOCKS
51 /*
52 * Ticket spinlocks are conceptually two atomic variables,
53 * one indicating the current FIFO head (spinlock owner),
54 * and the other indicating the current FIFO tail.
55 * Spinlock is acquired in the following manner:
56 * - current FIFO tail value is atomically incremented while it's
57 * original value is saved as a "ticket"
58 * - we spin until the FIFO head becomes equal to the ticket value
59 *
60 * Spinlock is released by atomic increment of the FIFO head
61 */
62 atomic_t owner;
63 atomic_t tail;
64#else
65 atomic_t locked;
66#endif /* CONFIG_TICKET_SPINLOCKS */
67#endif /* CONFIG_SMP */
68
69#ifdef CONFIG_SPIN_VALIDATE
70 /* Stores the thread that holds the lock with the locking CPU
71 * ID in the bottom two bits.
72 */
73 uintptr_t thread_cpu;
74#ifdef CONFIG_SPIN_LOCK_TIME_LIMIT
75 /* Stores the time (in cycles) when a lock was taken
76 */
77 uint32_t lock_time;
78#endif /* CONFIG_SPIN_LOCK_TIME_LIMIT */
79#endif /* CONFIG_SPIN_VALIDATE */
80
81#if defined(CONFIG_NONZERO_SPINLOCK_SIZE) && !defined(CONFIG_SMP) && !defined(CONFIG_SPIN_VALIDATE)
82 /* Add a dummy field to guarantee the spinlock has a non-zero
83 * size. If neither CONFIG_SMP nor CONFIG_SPIN_VALIDATE are
84 * defined then the k_spinlock struct would otherwise have no
85 * members and sizeof(k_spinlock) would be 0 in C and 1 in C++.
86 *
87 * That size difference causes problems when the k_spinlock
88 * is embedded into another struct like k_msgq, because C and
89 * C++ will have different ideas on the offsets of the members
90 * that come after the k_spinlock member.
91 */
92 char dummy;
93#endif
97};

181static ALWAYS_INLINE k_spinlock_key_t k_spin_lock(struct k_spinlock *l)
182{
183 ARG_UNUSED(l);
184 k_spinlock_key_t k;
185
186 /* Note that we need to use the underlying arch-specific lock
187 * implementation. The "irq_lock()" API in SMP context is
188 * actually a wrapper for a global spinlock!
189 */
190 k.key = arch_irq_lock();
191
192 z_spinlock_validate_pre(l);
193#ifdef CONFIG_SMP
194#ifdef CONFIG_TICKET_SPINLOCKS
195 /*
196 * Enqueue ourselves to the end of a spinlock waiters queue
197 * receiving a ticket
198 */
199 atomic_val_t ticket = atomic_inc(&l->tail);
200 /* Spin until our ticket is served */
201 while (atomic_get(&l->owner) != ticket) {
202 arch_spin_relax();
203 }
204#else
205 while (!atomic_cas(&l->locked, 0, 1)) {
206 do {
207 arch_spin_relax();
208 } while (atomic_get(&l->locked) != 0);
209 }
210#endif /* CONFIG_TICKET_SPINLOCKS */
211#endif /* CONFIG_SMP */
212 z_spinlock_validate_post(l);
213
214 return k;
215}

231static ALWAYS_INLINE int k_spin_trylock(struct k_spinlock *l, k_spinlock_key_t *k)
232{
233 int key = arch_irq_lock();
234
235 z_spinlock_validate_pre(l);
236#ifdef CONFIG_SMP
237#ifdef CONFIG_TICKET_SPINLOCKS
238 /*
239 * atomic_get and atomic_cas operations below are not executed
240 * simultaneously.
241 * So in theory k_spin_trylock can lock an already locked spinlock.
242 * To reproduce this the following conditions should be met after we
243 * executed atomic_get and before we executed atomic_cas:
244 *
245 * - spinlock needs to be taken 0xffff_..._ffff + 1 times
246 * (which requires 0xffff_..._ffff number of CPUs, as k_spin_lock call
247 * is blocking) or
248 * - spinlock needs to be taken and released 0xffff_..._ffff times and
249 * then taken again
250 *
251 * In real-life systems this is considered non-reproducible given that
252 * required actions need to be done during this tiny window of several
253 * CPU instructions (which execute with interrupt locked,
254 * so no preemption can happen here)
255 */
256 atomic_val_t ticket_val = atomic_get(&l->owner);
257
258 if (!atomic_cas(&l->tail, ticket_val, ticket_val + 1)) {
259 goto busy;
260 }
261#else
262 if (!atomic_cas(&l->locked, 0, 1)) {
263 goto busy;
264 }
265#endif /* CONFIG_TICKET_SPINLOCKS */
266#endif /* CONFIG_SMP */
267 z_spinlock_validate_post(l);
268
269 k->key = key;
270
271 return 0;
272
273#ifdef CONFIG_SMP
274busy:
275 arch_irq_unlock(key);
276 return -EBUSY;
277#endif /* CONFIG_SMP */
278}

301static ALWAYS_INLINE void k_spin_unlock(struct k_spinlock *l,
302 k_spinlock_key_t key)
303{
304 ARG_UNUSED(l);
305#ifdef CONFIG_SPIN_VALIDATE
306 __ASSERT(z_spin_unlock_valid(l), "Not my spinlock %p", l);
307
308#if defined(CONFIG_SPIN_LOCK_TIME_LIMIT) && (CONFIG_SPIN_LOCK_TIME_LIMIT != 0)
309 uint32_t delta = sys_clock_cycle_get_32() - l->lock_time;
310
311 __ASSERT(delta < CONFIG_SPIN_LOCK_TIME_LIMIT,
312 "Spin lock %p held %u cycles, longer than limit of %u cycles",
313 l, delta, CONFIG_SPIN_LOCK_TIME_LIMIT);
314#endif /* CONFIG_SPIN_LOCK_TIME_LIMIT */
315#endif /* CONFIG_SPIN_VALIDATE */
316
317#ifdef CONFIG_SMP
318#ifdef CONFIG_TICKET_SPINLOCKS
319 /* Give the spinlock to the next CPU in a FIFO */
320 (void)atomic_inc(&l->owner);
321#else
322 /* Strictly we don't need atomic_clear() here (which is an
323 * exchange operation that returns the old value). We are always
324 * setting a zero and (because we hold the lock) know the existing
325 * state won't change due to a race. But some architectures need
326 * a memory barrier when used like this, and we don't have a
327 * Zephyr framework for that.
328 */
329 (void)atomic_clear(&l->locked);
330#endif /* CONFIG_TICKET_SPINLOCKS */
331#endif /* CONFIG_SMP */
332 arch_irq_unlock(key.key);
333}

339#if defined(CONFIG_TEST) || defined(CONFIG_ASSERT)
340/*
341 * @brief Checks if spinlock is held by some CPU, including the local CPU.
342 * This should only be used in tests or assertions, not to make
343 * runtime control flow decisions.
344 *
345 * @param l A pointer to the spinlock
346 * @retval true - if spinlock is held by some CPU; false - otherwise
347 */
348static ALWAYS_INLINE bool z_spin_is_locked(struct k_spinlock *l)
349{
350#ifdef CONFIG_SMP
351#ifdef CONFIG_TICKET_SPINLOCKS
352 atomic_val_t ticket_val = atomic_get(&l->owner);
353
354 return !atomic_cas(&l->tail, ticket_val, ticket_val);
355#else
356 return l->locked;
357#endif /* CONFIG_TICKET_SPINLOCKS */
358#else
359 ARG_UNUSED(l);
360 /* In UP builds a spinlock reduces to an IRQ lock. */
361 return !arch_cpu_irqs_are_enabled();
362#endif /* CONFIG_SMP */
363}
364#endif /* defined(CONFIG_TEST) || defined(CONFIG_ASSERT) */
