78static inline double gptp_pow2(int exp)
79{
80 double res;
81
82 if (exp >= 0) {
83 res = 1 << exp;
84 } else {
85 res = 1.0;
86
87 while (exp++) {
88 res /= 2;
89 }
90 }
91
92 return res;
93}
