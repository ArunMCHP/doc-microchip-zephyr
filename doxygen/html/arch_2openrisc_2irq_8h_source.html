68static ALWAYS_INLINE unsigned int arch_irq_lock(void)
69{
70 const uint32_t sr = openrisc_read_spr(SPR_SR);
71
72 openrisc_write_spr(SPR_SR, sr & ~SPR_SR_IRQ_MASK);
73 return (sr & SPR_SR_IRQ_MASK) ? 1 : 0;
74}
75
76static ALWAYS_INLINE void arch_irq_unlock(unsigned int key)
77{
78 const uint32_t sr = openrisc_read_spr(SPR_SR);
79
80 openrisc_write_spr(SPR_SR, key ? (sr | SPR_SR_IRQ_MASK) : (sr & ~SPR_SR_IRQ_MASK));
81}

99static ALWAYS_INLINE void arch_irq_enable(unsigned int irq)
100{
101 const unsigned int key = arch_irq_lock();
102
103 openrisc_write_spr(SPR_PICMR, openrisc_read_spr(SPR_PICMR) | BIT(irq));
104 arch_irq_unlock(key);
105}

112static ALWAYS_INLINE void arch_irq_disable(unsigned int irq)
113{
114 const unsigned int key = arch_irq_lock();
115
116 openrisc_write_spr(SPR_PICMR, openrisc_read_spr(SPR_PICMR) & ~BIT(irq));
117 arch_irq_unlock(key);
118};

120static ALWAYS_INLINE int arch_irq_is_enabled(unsigned int irq)
